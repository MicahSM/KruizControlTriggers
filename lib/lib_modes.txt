########################
#  Chat Mode Changes   #
########################
#Only one at a time can be activated due to how timers work

onAction "EmoteOnlyMode"
chat send "/emoteonly"
delay {in1}
chat send "/emoteonlyoff"

onAction "SubOnlyMode"
chat send "/subcribers"
delay {in1}
chat send "/subscribersoff"

onAction "SlowMode"
chat send "/slow"
delay {in1}
chat send "/slowoff"
